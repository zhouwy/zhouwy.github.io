冒泡排序

// 冒泡排序
function bubbleSort(arr){
     for (var i = 0; i <arr.length; i++) {
          for (var j= 1; j < arr.length-i;j++) {
               if(arr[j-1]>arr[j]){
                    var tmp=arr[j-1];
                    arr[j-1]=arr[j];
                    arr[j]=tmp;
               }
          };
     };
     return arr;
}

// 快速排序
function quickSort(arr){
     if(arr.length<=1){
          return arr;
     }

     var pivot=arr[0];
     var left=[],
          right=[];
     for(var i=1;i<arr.length;i++){
          if(arr[i]<pivot){
               left.push(arr[i]);
          }else{
               right.push(arr[i]);
          }
     }

     return quickSort(left).concat(pivot,quickSort(right));
}

// 插入排序
function insertSort(arr){
     for (var i = 1; i < arr.length; i++) {
          var j=i;
          var key=arr[j];
          while(--j>-1){
               if(arr[j]>key){
                    arr[j+1]=arr[j];
               }else{
                    break;
               }
          }
          arr[j+1]=key;
     };
     return arr;
}
